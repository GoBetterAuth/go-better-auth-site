title: Basic Usage

```go title="main.go"
package main

import (
	"os"
	"log"
	"net/http"

	gobetterauth "github.com/GoBetterAuth/go-better-auth"
	gobetterauthdomain "github.com/GoBetterAuth/go-better-auth/domain"
)

func main() {
	auth, err := gobetterauth.New(&gobetterauthdomain.Config{
		AppName:  "My Application",
		BaseURL:  "http://localhost:8080"
		BasePath: "/api/auth",
		Database: gobetterauthdomain.DatabaseConfig{
			Provider:          "sqlite",
			ConnectionString: "./auth.db",
		},
		EmailAndPassword: &gobetterauthdomain.EmailPasswordConfig{
			Enabled:    true,
		},
	})
	if err != nil {
		log.Fatal(err)
	}

	// Mount auth routes
	http.Handle("/api/auth/", auth.Handler())

	// Add your application routes
	http.HandleFunc("/", func(w http.ResponseWriter, r *http.Request) {
		w.Write([]byte("Welcome to My App!"))
	})

	// Start the server
	log.Println("Server running on http://localhost:8080")
	log.Fatal(http.ListenAndServe(":8080", nil))
}
```

## Testing the Setup

### Sign Up

```bash
curl -X POST http://localhost:8080/api/auth/sign-up/email \
  -H "Content-Type: application/json" \
  -d '{"email": "john.doe@example.com", "password": "password123", "name": "John Doe"}'
```

### Sign In

```bash
curl -X POST http://localhost:8080/api/auth/sign-in/email \
  -H "Content-Type: application/json" \
  -d '{"email": "john.doe@example.com", "password": "password123"}'
```

### Get Session

```bash
curl -X GET http://localhost:8080/api/auth/me \
  -H "Cookie: go-better-auth.session_token=your-session-token"
```
